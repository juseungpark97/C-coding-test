#include <bits/stdc++.h>
using namespace std;

int k, cnt;
string a;
map<char, int> mp;

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    cout.tie(NULL);

    cin >> k;

    for(char i = 'a'; i <= 'z'; i++) mp.insert({i, 0});


    while(k--)
    {
        cin >> a;
        mp[a[0]]++;
    }

    for(char i = 'a'; i <= 'z'; i++)
    {
        if(mp[i] >= 5)
        {
            auto it = mp.find(i);
            cout << (*it).first;
            cnt++;
        } 
    }

    if(cnt == 0) cout << "PREDAJA";

    return 0;


}
