#include <bits/stdc++.h>
using namespace std;

string s, ret;
bool flag = true;

void check(string s1)
{
    if(s1 == "pi" || s1 == "ka") ret = "";
    else if(s1 == "ch") return;
    else if(s1 == "chu") ret = "";
    else flag = false;
}


int main()
{
    cin >> s;
    for(int i = 0; i < s.size(); i++)
    {
        ret += s[i];
        if(ret.size() >= 2) check(ret);
    }
    if(ret.size() == 0 && flag) cout << "YES";
    else cout << "NO";
}
