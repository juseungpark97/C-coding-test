#include <bits/stdc++.h>
using namespace std;

int psum[1000000], arr[1000000], n, k;
int sum = -10000001, temp;

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    cout.tie(NULL);
    cin >> n >> k;

    fill(psum, psum + 1000000, 0);


    for(int i = 1; i <= n; i++)
    {
        cin >> arr[i];
        psum[i] = psum[i-1] + arr[i];
    }

    if((n - k) == 0) cout << psum[k];
    else
    {
        for(int i = 0; i < n-k+1; i++)
    {
        temp = psum[k+i] - psum[i];
        sum = max(temp, sum);
    }
    cout << sum;


    }


    


}
